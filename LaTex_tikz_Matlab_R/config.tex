\KOMAoptions{
	paper			=	a4,									% layout for DIN A4
	twoside			=	on,									% layout for double sided printing
	fontsize		=	12pt,								% font size 12pt
	BCOR			=	12mm,								% correction for fixation	
	DIV				=	16,									% DIV-value for creating the print space	
	parskip			=	half,								% spacing between paragraphs							(maybe known as obsolete option halfparskip (obsolete))
	headinclude		=	true,								% include head in layout computation
	headings		=	openright,							% chapters start odd pages	
	captions		=	tableheading,						% use table headings with correct spacing				(maybe known as option tablecaptionabove (obsolete))
	listof			=	totoc,								% this option will add the table of images and table of tables to the table of contents
	numbers			=	noenddot,							% chapter numbers without full stop						(maybe known as option pointlessnumbers (obsolete))
	bibliography	=	totoc,								% put bibliography into table of contents				(maybe known as option bibtotoc (obsolete))
}

% ==============================================
% =============== caption style ================
% ==============================================

\captionsetup{												% for figures and tables
	font			=	small,								% set font size for caption and text (use labelfont and textfont for seperate or exclusive settings)
	textfont		=	it,									% name font as italic
	labelfont		=	bf,									% set the label font to a specific format (bf = boldfont)
	justification	=	justified,							% caption will use justification format
	format			=	plain,								% specify if the text should start under the caption name or not (try hang to see the difference)
	indention		=	.5cm,								% use an extra indention starting at the second line of the caption
	labelformat		=	default,							% other options are empty, brace, simple and parens
	labelsep		=	space,								% other options are none, colon, period, quad (bigger space), newline and endash
	skip			=	10pt,								% set the vertical space between the caption and the figure or table contents (default 10pt)
	position		=	bottom,								% assume that the caption is standing at the bottom of the environment and ther		\glsentryitem{##1}\glstarget{##1}{\glossentryname{##1}} &efore the space (skip=<amount>) is placed above
	listformat		=	simple								% specify the list format for \listoffigures	
}
